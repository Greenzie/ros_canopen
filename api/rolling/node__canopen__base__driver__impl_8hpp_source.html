124 template <class NODETYPE>
125 void NodeCanopenBaseDriver<NODETYPE>::add_to_master()
126 {
127  RCLCPP_INFO(this->node_->get_logger(), "eds file %s", this->eds_.c_str());
128  RCLCPP_INFO(this->node_->get_logger(), "bin file %s", this->bin_.c_str());
129  std::shared_ptr<std::promise<std::shared_ptr<ros2_canopen::LelyDriverBridge>>> prom;
130  prom = std::make_shared<std::promise<std::shared_ptr<ros2_canopen::LelyDriverBridge>>>();
131  std::future<std::shared_ptr<ros2_canopen::LelyDriverBridge>> f = prom->get_future();
132  this->exec_->post(
133  [this, prom]()
134  {
135  std::scoped_lock<std::mutex> lock(this->driver_mutex_);
136  this->lely_driver_ = std::make_shared<ros2_canopen::LelyDriverBridge>(
137  *(this->exec_), *(this->master_), this->node_id_, this->node_->get_name(), this->eds_,
138  this->bin_);
139  this->driver_ = std::static_pointer_cast<lely::canopen::BasicDriver>(this->lely_driver_);
140  prom->set_value(lely_driver_);
141  });
142 
143  auto future_status = f.wait_for(this->non_transmit_timeout_);
144  if (future_status != std::future_status::ready)
145  {
146  RCLCPP_ERROR(this->node_->get_logger(), "Adding timed out.");
147  throw DriverException("add_to_master: adding timed out");
148  }
149  this->lely_driver_ = f.get();
150  this->driver_ = std::static_pointer_cast<lely::canopen::BasicDriver>(this->lely_driver_);
151  if (!this->lely_driver_->IsReady())
152  {
153  RCLCPP_WARN(this->node_->get_logger(), "Wait for device to boot.");
154  try
155  {
156  this->lely_driver_->wait_for_boot();
157  }
158  catch (const std::exception & e)
159  {
160  RCLCPP_ERROR(this->node_->get_logger(), e.what());
161  }
162  }
163  RCLCPP_INFO(this->node_->get_logger(), "Driver booted and ready.");
164 }

185 template <class NODETYPE>
186 void NodeCanopenBaseDriver<NODETYPE>::nmt_listener()
187 {
188  while (rclcpp::ok())
189  {
190  std::future<lely::canopen::NmtState> f;
191  {
192  std::scoped_lock<std::mutex> lock(this->driver_mutex_);
193  f = this->lely_driver_->async_request_nmt();
194  }
195  while (f.wait_for(this->non_transmit_timeout_) != std::future_status::ready)
196  {
197  if (!this->activated_.load()) return;
198  }
199  try
200  {
201  auto state = f.get();
202  if (nmt_state_cb_)
203  {
204  nmt_state_cb_(state, this->lely_driver_->get_id());
205  }
206  on_nmt(state);
207  }
208  catch (const std::future_error & e)
209  {
210  break;
211  }
212  }
